# Forensics4Linux
 Scripts for Linux Forensics.  
 Output to a formatted .txt file for further analysis.

# Install
git clone https://github.com/ECEStuffNWJ/Forensics4Linux.git

# Run
cd Forensics4Linux   
chmod +x linfor.sh    
./linfor.sh 

# Results
cd /tmp  
look for Forensics4Linux.txt
